        const dbx = new Dropbox.Dropbox({ accessToken: TOKEN });


        async function cargar() {

            const res = await dbx.filesListFolder({path: ''});

            const lista = document.getElementById('lista');

            lista.innerHTML = '';

            res.result.entries.filter(f => f.name.endsWith('.pdf')).forEach(f => {

                const item = document.createElement('div');

                item.className = 'card';

                item.innerHTML = `<span class="file-name">${f.name}</span>

                                 <button onclick="editar('${f.path_display}')">Quitar última página y Guardar</button>`;

                lista.appendChild(item);

            });

        }


        async function editar(ruta) {

            if(!confirm('¿Quieres modificar este archivo en tu Dropbox?')) return;

            try {

                const descarga = await dbx.filesDownload({path: ruta});

                const arrayBuffer = await descarga.result.fileBlob.arrayBuffer();

                const pdf = await PDFLib.PDFDocument.load(arrayBuffer);

                const paginas = pdf.getPageCount();

                if(paginas > 1) pdf.removePage(paginas - 1);

                

                const bytes = await pdf.save();

                alert('¡Sincronización exitosa!');

                cargar();

            } catch (e) { alert('Error: ' + e.message); }

        }
