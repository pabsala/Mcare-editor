        const dbx = new Dropbox.Dropbox({ accessToken: ACCESS_TOKEN });


        async function listarArchivos() {

            const status = document.getElementById('status');

            const fileList = document.getElementById('file-list');

            

            try {

                const response = await dbx.filesListFolder({ path: '' });

                status.style.display = 'none';

                fileList.innerHTML = '';


                // Filtramos solo los archivos PDF

                const archivos = response.result.entries.filter(entry => entry.name.toLowerCase().endsWith('.pdf'));


                if (archivos.length === 0) {

                    status.style.display = 'block';

                    status.innerText = 'No se encontraron archivos PDF en la raíz.';

                    return;

                }


                archivos.forEach(file => {

                    const card = document.createElement('div');

                    card.className = 'card';

                    card.innerHTML = `

                        <span class="file-name">${file.name}</span>

                        <button onclick="editarPDF('${file.path_display}')">Quitar última página</button>

                    `;

                    fileList.appendChild(card);

                });

            } catch (error) {

                status.innerText = 'Error de conexión: Verifica tu Token de Dropbox.';

                console.error(error);

            }

        }


        async function editarPDF(path) {

            if (!confirm('¿Seguro que quieres eliminar la última página de este archivo?')) return;

            

            try {

                const btn = event.target;

                const textoOriginal = btn.innerText;

                btn.innerText = 'Procesando...';

                btn.disabled = true;


                // 1. Descargar archivo

                const response = await dbx.filesDownload({ path: path });

                const blob = response.result.fileBlob;

                const arrayBuffer = await blob.arrayBuffer();


                // 2. Modificar con PDF-Lib

                const pdfDoc = await PDFLib.PDFDocument.load(arrayBuffer);

                const pageCount = pdfDoc.getPageCount();


                if (pageCount <= 1) {

                    alert('El documento solo tiene una página, no se puede eliminar.');

                } else {

                    pdfDoc.removePage(pageCount - 1);

                    const pdfBytes = await pdfDoc.save();


                    // 3. Subir de vuelta (Sobrescribir)

                    await dbx.filesUpload({

                        path: path,

                        contents: pdfBytes,

                        mode: 'overwrite'

                    });

                    alert('¡Actualizado con éxito!');

                }

                

                btn.innerText = textoOriginal;

                btn.disabled = false;

                listarArchivos();

            } catch (error) {

                alert('Ocurrió un error al editar.');

                console.error(error);

            }

        }


        // Iniciar la carga

        listarArchivos();
